# 算法
## 链表
`链表逆置`
```
	function nizhi(root) {
		if (root.next.next = null) {
			root.next.next = root
			return root.next
		}
		let result = nizhi(root.next);
		root.next.next = root;
		root.next = null;
		return result;
	}
```
## 数组
`冒泡排序`
```
    // 比较
    function compare(a, b) {
      return a > b
    }

	// 排序
    function sort(arr) {
      let a, b;
      for (let j = 0; j < arr.length; j++) {
        for (let i = 0; i < arr.length - 1 - j; i++) {
          a = arr[i];
          b = arr[i + 1];
          if (compare(a, b)) {
            arr[i] = b;
            arr[i + 1] = a;
          }
        }
      }
      return arr;
    }
```
